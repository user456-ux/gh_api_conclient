
using System.Net.Http.Json;

// Use the actual URL from your running API
string apiUrl1 = "http://localhost:5257/weatherforecast";  //no controller name, just the endpoint
string apiUrl2 = "http://localhost:5000/weatherforecast";

using HttpClient client = new();

try
{
    // Fetch and deserialize JSON directly into a list
    var forecasts1 = await client.GetFromJsonAsync<List<WeatherForecast>>(apiUrl1);
    var forecasts2 = await client.GetStringAsync(apiUrl2);
    Console.WriteLine($"data: {forecasts2} \r\n");

    if (forecasts1 != null) { 
        foreach (var f in forecasts1)
        {
            Console.WriteLine($"{f.Date}: {f.TemperatureC}°C ({f.Summary})");
        }
    }

}
catch (HttpRequestException e)
{
    Console.WriteLine($"Request error: {e.Message}");
}

public class WeatherForecast
{
    public DateOnly Date { get; set; }
    public int TemperatureC { get; set; }
    public int TemperatureF => 32 + (int)(TemperatureC / 0.5556);
    public string? Summary { get; set; }
}
